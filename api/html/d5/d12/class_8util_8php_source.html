217 public static function clearFolder($path, $exclude = array())
218 {
219 $result = array();
220 $result['return'] = true;
221 $result['nb_files'] = 0;
222
223 $handle = @opendir($path);
224 if (!$handle) {
225 return null;
226 }
227
228 while (false !== ($file = readdir($handle))) {
229 if ($file == '.' || $file == '..' || in_array($file, $exclude)) {
230 continue;
231 }
232 if (is_dir($path . '/' . $file)) {
233 $r = self::clearFolder($path . '/' . $file);
234 if (!$r) {
235 $result['return'] = false;
236 return $result;
237 }
238 } else {
239 $r = @unlink($path . '/' . $file);
240 if ($r) {
241 $result['nb_files']++;
242 } else {
243 $result['return'] = false;
244 return $result;
245 }
246 }
247 }
248
249 closedir($handle);
250 return $result;
251 }

354 public static function replaceInFile($path, $replaceList)
355 {
356 if (file_exists($path)) {
357 $lines = file($path);
358 $fp = fopen($path, 'w');
359 foreach ($lines as $nb => $line) {
360 $replaceDone = false;
361 foreach ($replaceList as $regex => $replace) {
362 if (preg_match($regex, $line, $matches)) {
363 $countParams = preg_match_all('/{{(\d+)}}/', $replace, $paramsMatches);
364 if ($countParams > 0 && $countParams <= count($matches)) {
365 foreach ($paramsMatches[1] as $paramsMatch) {
366 $replace = str_replace('{{' . $paramsMatch . '}}', $matches[$paramsMatch], $replace);
367 }
368 }
369 self::logTrace('Replace in file ' . $path . ' :');
370 self::logTrace('## line_num: ' . trim($nb));
371 self::logTrace('## old: ' . trim($line));
372 self::logTrace('## new: ' . trim($replace));
373 fwrite($fp, $replace . PHP_EOL);
374
375 $replaceDone = true;
376 break;
377 }
378 }
379 if (!$replaceDone) {
380 fwrite($fp, $line);
381 }
382 }
383 fclose($fp);
384 }

798 public static function findRepos($initPath, $startPath, $checkFile, $maxDepth = 1)
799 {
800 $depth = substr_count(str_replace($initPath, '', $startPath), '/');
801 $result = array();
802
803 $handle = @opendir($startPath);
804 if (!$handle) {
805 return $result;
806 }
807
808 while (false !== ($file = readdir($handle))) {
809 if ($file == '.' || $file == '..') {
810 continue;
811 }
812 if (is_dir($startPath . '/' . $file) && ($initPath == $startPath || $depth <= $maxDepth)) {
813 $tmpResults = self::findRepos($initPath, $startPath . '/' . $file, $checkFile, $maxDepth);
814 foreach ($tmpResults as $tmpResult) {
815 $result[] = $tmpResult;
816 }
817 } elseif (is_file($startPath . '/' . $checkFile) && !in_array($startPath, $result)) {
818 $result[] = self::formatUnixPath($startPath);
819 }
820 }
821
822 closedir($handle);
823 return $result;
824 }

1141 private static function findFiles($startPath, $includes = array(''), $recursive = true)
1142 {
1143 $result = array();
1144
1145 $handle = @opendir($startPath);
1146 if (!$handle) {
1147 return $result;
1148 }
1149
1150 while (false !== ($file = readdir($handle))) {
1151 if ($file == '.' || $file == '..') {
1152 continue;
1153 }
1154 if (is_dir($startPath . '/' . $file) && $recursive) {
1155 $tmpResults = self::findFiles($startPath . '/' . $file, $includes);
1156 foreach ($tmpResults as $tmpResult) {
1157 $result[] = $tmpResult;
1158 }
1159 } elseif (is_file($startPath . '/' . $file)) {
1160 foreach ($includes as $include) {
1161 if (self::startWith($include, '!')) {
1162 $include = ltrim($include, '!');
1163 if (self::startWith($file, '.') && !self::endWith($file, $include)) {
1164 $result[] = self::formatUnixPath($startPath . '/' . $file);
1165 } elseif ($file != $include) {
1166 $result[] = self::formatUnixPath($startPath . '/' . $file);
1167 }
1168 } elseif (self::endWith($file, $include) || $file == $include || empty($include)) {
1169 $result[] = self::formatUnixPath($startPath . '/' . $file);
1170 }
1171 }
1172 }
1173 }
1174
1175 closedir($handle);
1176 return $result;
