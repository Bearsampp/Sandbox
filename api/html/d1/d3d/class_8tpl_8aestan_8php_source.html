218 private static function getActionService($service, $action, $item = false)
219 {
220 global $bearsamppLang;
221 $result = 'Action: ' . $action;
222
223 if ($service != null) {
224 $result = 'Action: service; ' .
225 'Service: ' . $service . '; ' .
226 'ServiceAction: ' . $action;
227 }
228
229 if ($item) {
230 $result = 'Type: item; ' . $result;
231 if ($action == self::SERVICE_START) {
232 $result .= '; Caption: "' . $bearsamppLang->getValue(Lang::MENU_START_SERVICE) . '"' .
233 '; Glyph: ' . self::GLYPH_START;
234 } elseif ($action == self::SERVICE_STOP) {
235 $result .= '; Caption: "' . $bearsamppLang->getValue(Lang::MENU_STOP_SERVICE) . '"' .
236 '; Glyph: ' . self::GLYPH_STOP;
237 } elseif ($action == self::SERVICE_RESTART) {
238 $result .= '; Caption: "' . $bearsamppLang->getValue(Lang::MENU_RESTART_SERVICE) . '"' .
239 '; Glyph: ' . self::GLYPH_RELOAD;
240 }
241 } elseif ($action != self::SERVICES_CLOSE) {
242 $result .= '; Flags: ignoreerrors waituntilterminated';
243 }
244
245 return $result;
246 }
