145 private static function getActionService($service, $action, $item = false)
146 {
147 global $bearsamppLang;
148 $result = 'Action: ' . $action;
149
150 if ($service != null) {
151 $result = 'Action: service; ' .
152 'Service: ' . $service . '; ' .
153 'ServiceAction: ' . $action;
154 }
155
156 if ($item) {
157 $result = 'Type: item; ' . $result;
158 if ($action == self::SERVICE_START) {
159 $result .= '; Caption: "' . $bearsamppLang->getValue(Lang::MENU_START_SERVICE) . '"' .
160 '; Glyph: ' . self::GLYPH_START;
161 } elseif ($action == self::SERVICE_STOP) {
162 $result .= '; Caption: "' . $bearsamppLang->getValue(Lang::MENU_STOP_SERVICE) . '"' .
163 '; Glyph: ' . self::GLYPH_STOP;
164 } elseif ($action == self::SERVICE_RESTART) {
165 $result .= '; Caption: "' . $bearsamppLang->getValue(Lang::MENU_RESTART_SERVICE) . '"' .
166 '; Glyph: ' . self::GLYPH_RELOAD;
167 }
168 } elseif ($action != self::SERVICES_CLOSE) {
169 $result .= '; Flags: ignoreerrors waituntilterminated';
170 }
171
172 return $result;
173 }
