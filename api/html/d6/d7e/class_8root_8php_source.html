526 /**
527 * Handles errors and logs them to the error log file.
528 *
529 * @param int $errno The level of the error raised.
530 * @param string $errstr The error message.
531 * @param string $errfile The filename that the error was raised in.
532 * @param int $errline The line number the error was raised at.
533 */
534 public function errorHandler($errno, $errstr, $errfile, $errline)
535 {
536 if (error_reporting() === 0) {
537 return;
538 }
539
540 $errfile = Util::formatUnixPath($errfile);
541 $errfile = str_replace($this->getRootPath(), '', $errfile);
542
543 if (!defined('E_DEPRECATED')) {
544 define('E_DEPRECATED', 8192);
545 }
546
547 $errNames = array(
548 E_ERROR => 'E_ERROR',
549 E_WARNING => 'E_WARNING',
550 E_PARSE => 'E_PARSE',
551 E_NOTICE => 'E_NOTICE',
552 E_CORE_ERROR => 'E_CORE_ERROR',
553 E_CORE_WARNING => 'E_CORE_WARNING',
554 E_COMPILE_ERROR => 'E_COMPILE_ERROR',
555 E_COMPILE_WARNING => 'E_COMPILE_WARNING',
556 E_USER_ERROR => 'E_USER_ERROR',
557 E_USER_WARNING => 'E_USER_WARNING',
558 E_USER_NOTICE => 'E_USER_NOTICE',
559 E_STRICT => 'E_STRICT',
560 E_RECOVERABLE_ERROR => 'E_RECOVERABLE_ERROR',
561 E_DEPRECATED => 'E_DEPRECATED',
562 );
563
564 $content = '[' . date('Y-m-d H:i:s', time()) . '] ';
565 $content .= $errNames[$errno] . ' ';
566 $content .= $errstr . ' in ' . $errfile;
567 $content .= ' on line ' . $errline . PHP_EOL;
568 $content .= self::debugStringBacktrace() . PHP_EOL;
569
570 file_put_contents($this->getErrorLogFilePath(), $content, FILE_APPEND);
571 }

573 /**
574 * Generates a debug backtrace string.
575 *
576 * @return string The debug backtrace.
577 */
578 private static function debugStringBacktrace()
579 {
580 ob_start();
581 debug_print_backtrace();
582 $trace = ob_get_contents();
583 ob_end_clean();
584
585 $trace = preg_replace('/^#0\s+Root::debugStringBacktrace[^\n]*\n/', '', $trace, 1);
586 $trace = preg_replace('/^#1\s+isRoot->errorHandler[^\n]*\n/', '', $trace, 1);
587 $trace = preg_replace_callback('/^#(\d+)/m', 'debugStringPregReplace', $trace);
588 return $trace;
589 }
