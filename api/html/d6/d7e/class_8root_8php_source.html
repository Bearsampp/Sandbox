535 public function errorHandler($errno, $errstr, $errfile, $errline)
536 {
537 if (error_reporting() === 0) {
538 return;
539 }
540
541 $errfile = Util::formatUnixPath($errfile);
542 $errfile = str_replace($this->getRootPath(), '', $errfile);
543
544 if (!defined('E_DEPRECATED')) {
545 define('E_DEPRECATED', 8192);
546 }
547
548 $errNames = array(
549 E_ERROR => 'E_ERROR',
550 E_WARNING => 'E_WARNING',
551 E_PARSE => 'E_PARSE',
552 E_NOTICE => 'E_NOTICE',
553 E_CORE_ERROR => 'E_CORE_ERROR',
554 E_CORE_WARNING => 'E_CORE_WARNING',
555 E_COMPILE_ERROR => 'E_COMPILE_ERROR',
556 E_COMPILE_WARNING => 'E_COMPILE_WARNING',
557 E_USER_ERROR => 'E_USER_ERROR',
558 E_USER_WARNING => 'E_USER_WARNING',
559 E_USER_NOTICE => 'E_USER_NOTICE',
560 E_STRICT => 'E_STRICT',
561 E_RECOVERABLE_ERROR => 'E_RECOVERABLE_ERROR',
562 E_DEPRECATED => 'E_DEPRECATED',
563 );
564
565 $content = '[' . date('Y-m-d H:i:s', time()) . '] ';
566 $content .= $errNames[$errno] . ' ';
567 $content .= $errstr . ' in ' . $errfile;
568 $content .= ' on line ' . $errline . PHP_EOL;
569 $content .= self::debugStringBacktrace() . PHP_EOL;
570
571 file_put_contents($this->getErrorLogFilePath(), $content, FILE_APPEND);
572 }

579 private static function debugStringBacktrace()
580 {
581 ob_start();
582 debug_print_backtrace();
583 $trace = ob_get_contents();
584 ob_end_clean();
585
586 $trace = preg_replace('/^#0\s+Root::debugStringBacktrace[^\n]*\n/', '', $trace, 1);
587 $trace = preg_replace('/^#1\s+isRoot->errorHandler[^\n]*\n/', '', $trace, 1);
588 $trace = preg_replace_callback('/^#(\d+)/m', 'debugStringPregReplace', $trace);
589 return $trace;
