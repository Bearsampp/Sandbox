98 public function status($timeout = true)
99 {
100 usleep(self::SLEEP_TIME);
101
102 $this->latestStatus = self::STATUS_NA;
103 $maxtime = time() + self::PENDING_TIMEOUT;
104
105 while ($this->latestStatus == self::STATUS_NA || $this->isPending($this->latestStatus)) {
106 $exec = $this->exec('status ' . $this->getName());
107 if ($exec !== false) {
108 if (count($exec) > 1) {
109 $this->latestStatus = self::STATUS_NOT_EXIST;
110 } else {
111 $this->latestStatus = $exec[0];
112 }
113 }
114 if ($timeout && $maxtime < time()) {
115 break;
116 }
117 }
118
119 if ($this->latestStatus == self::STATUS_NOT_EXIST) {
120 $this->latestError = 'Error 3: The specified service does not exist as an installed service.';
121 $this->latestStatus = self::STATUS_NA;
122 }
123
124 return $this->latestStatus;
