79 private function exec($args)
80 {
81 global $bearsamppCore;
82
83 $command = '"' . $bearsamppCore->getNssmExe() . '" ' . $args;
84 $this->writeLogInfo('Cmd: ' . $command);
85
86 $result = Batch::exec('nssm', $command, 10);
87 if (is_array($result)) {
88 $rebuildResult = array();
89 foreach ($result as $row) {
90 $row = trim($row);
91 if (!empty($row)) {
92 $rebuildResult[] = preg_replace('/[\x00-\x1F\x80-\xFF]/', '', $row);
93 }
94 }
95 $result = $rebuildResult;
96 if (count($result) > 1) {
97 $this->latestError = implode(' ; ', $result);
98 }
99 return $result;
100 }
101
102 return false;
103 }

105 public function status($timeout = true)
106 {
107 usleep(self::SLEEP_TIME);
108
109 $this->latestStatus = self::STATUS_NA;
110 $maxtime = time() + self::PENDING_TIMEOUT;
111
112 while ($this->latestStatus == self::STATUS_NA || $this->isPending($this->latestStatus)) {
113 $exec = $this->exec('status ' . $this->getName());
114 if ($exec !== false) {
115 if (count($exec) > 1) {
116 $this->latestStatus = self::STATUS_NOT_EXIST;
117 } else {
118 $this->latestStatus = $exec[0];
119 }
120 }
121 if ($timeout && $maxtime < time()) {
122 break;
123 }
124 }
125
126 if ($this->latestStatus == self::STATUS_NOT_EXIST) {
127 $this->latestError = 'Error 3: The specified service does not exist as an installed service.';
128 $this->latestStatus = self::STATUS_NA;
129 }
130
131 return $this->latestStatus;
132 }

134 public function create()
135 {
136 $this->writeLog('Create service');

144 $this->writeLog('-> environment extra: ' . $this->getEnvironmentExtra());
145 $this->writeLog('-> start_type: ' . ($this->getStart() != null ? $this->getStart() : self::SERVICE_DEMAND_START));
146
147 // Install bin
148 $exec = $this->exec('install ' . $this->getName() . ' "' . $this->getBinPath() . '"');

152
153 // Params

158
159 // DisplayName
160 $exec = $this->exec('set ' . $this->getName() . ' DisplayName "' . $this->getDisplayName() . '"');
161 if ($exec === false) {
162 return false;
163 }
164
165 // Description
166 $exec = $this->exec('set ' . $this->getName() . ' Description "' . $this->getDisplayName() . '"');
167 if ($exec === false) {
168 return false;
169 }
170
171 // No AppNoConsole to fix nssm problems with Windows 10 Creators update.
172 $exec = $this->exec('set ' . $this->getName() . ' AppNoConsole "1"');
173 if ($exec === false) {
174 return false;
175 }
176
177 // Start
178 $exec = $this->exec('set ' . $this->getName() . ' Start "' . ($this->getStart() != null ? $this->getStart() : self::SERVICE_DEMAND_START) . '"');
179 if ($exec === false) {
180 return false;
181 }
182
183 // Stdout
184 $exec = $this->exec('set ' . $this->getName() . ' AppStdout "' . $this->getStdout() . '"');
185 if ($exec === false) {
186 return false;
187 }
188
189 // Stderr
190 $exec = $this->exec('set ' . $this->getName() . ' AppStderr "' . $this->getStderr() . '"');
191 if ($exec === false) {
192 return false;
193 }
194
195 // Environment Extra
196 $exec = $this->exec('set ' . $this->getName() . ' AppEnvironmentExtra ' . $this->getEnvironmentExtra());
197 if ($exec === false) {
198 return false;
199 }
200
201 if (!$this->isInstalled()) {
202 $this->latestError = null;
203 return false;
204 }
205
206 return true;

269 public function infos()
270 {
271 global $bearsamppRegistry;
272
273 $infos = Vbs::getServiceInfos($this->getName());
274 if ($infos === false) {
275 return false;
276 }
277
278 $infosNssm = array();
279 $infosKeys = array(
280 self::INFO_APPLICATION,
281 self::INFO_APP_PARAMETERS,
282 );
283
284 foreach ($infosKeys as $infoKey) {
285 $value = null;
286 $exists = $bearsamppRegistry->exists(
287 Registry::HKEY_LOCAL_MACHINE,
288 'SYSTEM\CurrentControlSet\Services\\' . $this->getName() . '\Parameters',
289 $infoKey
290 );
291 if ($exists) {
292 $value = $bearsamppRegistry->getValue(
293 Registry::HKEY_LOCAL_MACHINE,
294 'SYSTEM\CurrentControlSet\Services\\' . $this->getName() . '\Parameters',
295 $infoKey
296 );
297 }
298 $infosNssm[$infoKey] = $value;
299 }
300
301 if (!isset($infosNssm[self::INFO_APPLICATION])) {
302 return $infos;
303 }
304
305 $infos[Win32Service::VBS_PATH_NAME] = $infosNssm[Nssm::INFO_APPLICATION] . ' ' . $infosNssm[Nssm::INFO_APP_PARAMETERS];
306 return $infos;
