101 protected function replaceAll($params) {
102 $content = file_get_contents($this->bearsamppConf);
103
104 foreach ($params as $key => $value) {
105 $content = preg_replace('|' . $key . ' = .*|', $key . ' = ' . '"' . $value.'"', $content);
106 $this->bearsamppConfRaw[$key] = $value;
107 switch ($key) {
108 case self::LOCAL_CFG_API_PORT:
109 $this->apiPort = intval($value);
110 break;
111 case self::LOCAL_CFG_UI_PORT:
112 $this->uiPort = intval($value);
113 break;
114 case self::LOCAL_CFG_SMTP_PORT:
115 $this->smtpPort = intval($value);
116 break;
117 }
118 }
119
120 file_put_contents($this->bearsamppConf, $content);
121 }
