128 public static function getMenuPhpSettings()
129 {
130 global $bearsamppBins;
131
132 $menuItems = '';
133 $menuActions = '';
134 foreach ($bearsamppBins->getPhp()->getSettings() as $key => $value) {
135 if (is_array($value)) {
136 $menuItems .= 'Type: submenu; ' .
137 'Caption: "' . $key . '"; ' .
138 'SubMenu: MenuPhpSetting-' . md5($key) . '; ' .
139 'Glyph: ' . TplAestan::GLYPH_FOLDER_CLOSE . PHP_EOL;
140 } else {
141 $glyph = '';
142 $settingEnabled = $bearsamppBins->getPhp()->isSettingActive($value);
143 if (!$bearsamppBins->getPhp()->isSettingExists($value)) {
144 $glyph = TplAestan::GLYPH_WARNING;
145 } elseif ($settingEnabled) {
146 $glyph = TplAestan::GLYPH_CHECK;
147 }
148 $tplSwitchPhpSetting = TplApp::getActionMulti(
149 self::ACTION_SWITCH_SETTING, array($value, $settingEnabled),
150 array($key, $glyph),
151 false, get_called_class()
152 );
153
154 $menuItems .= $tplSwitchPhpSetting[TplApp::SECTION_CALL] . PHP_EOL;
155 $menuActions .= $tplSwitchPhpSetting[TplApp::SECTION_CONTENT];
156 }
157 }
158
159 $submenusItems = '';
160 $submenusActions = '';
161 $submenuKeys = self::getSubmenuPhpSettings();
162 foreach ($submenuKeys as $submenuKey) {
163 $submenusItems .= PHP_EOL . '[MenuPhpSetting-' . md5($submenuKey) . ']' .
164 PHP_EOL . self::getSubmenuPhpSettings($submenuKey);
165
166 $submenusActions .= self::getSubmenuPhpSettings($submenuKey, array(), array(), false);
167 }
168
169 return $menuItems . $submenusItems . PHP_EOL . $menuActions . $submenusActions;
170 }

172 private static function getSubmenuPhpSettings($passThr = array(), $result = array(), $settings = array(), $sectionCall = true)
173 {
174 global $bearsamppBins;
175 $settings = empty($settings) ? $bearsamppBins->getPhp()->getSettings() : $settings;
176
177 foreach ($settings as $key => $value) {
178 if (is_array($value)) {
179 if (is_array($passThr)) {
180 array_push($result, $key);
181 $result = self::getSubmenuPhpSettings($passThr, $result, $value);
182 } else {
183 $result = is_array($result) ? '' : $result;
184 if ($key == $passThr) {
185 foreach ($value as $key2 => $value2) {
186 if (is_array($value2) && $sectionCall) {
187 $result .= 'Type: submenu; ' .
188 'Caption: "' . $key2 . '"; ' .
189 'SubMenu: MenuPhpSetting-' . md5($key2) . '; ' .
190 'Glyph: ' . TplAestan::GLYPH_FOLDER_CLOSE . PHP_EOL;
191 } elseif (!is_array($value2)) {
192 $glyph = '';
193 $settingEnabled = $bearsamppBins->getPhp()->isSettingActive($value2);
194 if (!$bearsamppBins->getPhp()->isSettingExists($value2)) {
195 $glyph = TplAestan::GLYPH_WARNING;
196 } elseif ($settingEnabled) {
197 $glyph = TplAestan::GLYPH_CHECK;
198 }
199 $tplSwitchPhpSetting = TplApp::getActionMulti(
200 self::ACTION_SWITCH_SETTING, array($value2, $settingEnabled),
201 array($key2, $glyph),
202 false, get_called_class()
203 );
204
205 if ($sectionCall) {
206 $result .= $tplSwitchPhpSetting[TplApp::SECTION_CALL] . PHP_EOL;
207 } else {
208 $result .= $tplSwitchPhpSetting[TplApp::SECTION_CONTENT] . PHP_EOL;
209 }
210 }
211 }
212 } else {
213 $result .= self::getSubmenuPhpSettings($passThr, null, $value, $sectionCall);
214 }
215 }
216 }
217 }
218
219 return $result;
