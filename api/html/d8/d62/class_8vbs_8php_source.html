155 if (is_array($result) && count($result) > 0) {
156 $rebuildResult = array();
157 foreach ($result as $row) {
158 $row = explode(trim(self::STR_SEPARATOR), $row);
159 if (count($row) != count($vbsKeys)) {
160 continue;
161 }
162 $processInfo = array();
163 foreach ($vbsKeys as $key => $vbsKey) {
164 $processInfo[$vbsKey] = trim($row[$key]);
165 }
166 if (!empty($processInfo[Win32Ps::EXECUTABLE_PATH])) {
167 $rebuildResult[] = $processInfo;
168 }
169 }

319 public static function exec($basename, $resultFile, $content, $timeout = true)
320 {
321 global $bearsamppConfig, $bearsamppWinbinder;
322 $result = false;
323
324 $scriptPath = self::getTmpFile('.vbs', $basename);
325 $checkFile = self::getTmpFile('.tmp', $basename);
326 $errFile = self::getTmpFile('.tmp', $basename);
327 $randomVarName = Util::random(15, false);
328 $randomObjErrFile = Util::random(15, false);
329 $randomObjFile = Util::random(15, false);
330 $randomObjFso = Util::random(15, false);
331
332 // Header
333 $header = 'On Error Resume Next' . PHP_EOL .
334 'Dim ' . $randomVarName . ', ' . $randomObjFso . ', ' . $randomObjErrFile . ', ' . $randomObjFile . PHP_EOL .
335 'Set ' . $randomObjFso . ' = CreateObject("scripting.filesystemobject")' . PHP_EOL .
336 'Set ' . $randomObjErrFile . ' = ' . $randomObjFso . '.CreateTextFile("' . $errFile . '", True)' . PHP_EOL .
337 'Set ' . $randomObjFile . ' = ' . $randomObjFso . '.CreateTextFile("' . $checkFile . '", True)' . PHP_EOL . PHP_EOL;
338
339 // Footer
340 $footer = PHP_EOL . PHP_EOL .
341 'If Err.Number <> 0 Then' . PHP_EOL .
342 $randomObjErrFile . '.Write Err.Description' . PHP_EOL .
343 'End If' . PHP_EOL .
344 $randomObjFile . '.Write "' . self::END_PROCESS_STR . '"' . PHP_EOL .
345 $randomObjFile . '.Close' . PHP_EOL .
346 $randomObjErrFile . '.Close' . PHP_EOL;
347
348 // Process
349 file_put_contents($scriptPath, $header . $content . $footer);
350 $bearsamppWinbinder->exec('wscript.exe', '"' . $scriptPath . '"');
351
352 $timeout = is_numeric($timeout) ? $timeout : ($timeout === true ? $bearsamppConfig->getScriptsTimeout() : false);
353 $maxtime = time() + $timeout;
354 $noTimeout = $timeout === false;
355 while ($result === false || empty($result)) {
356 if (file_exists($checkFile)) {
357 $check = file($checkFile);
358 if (!empty($check) && trim($check[0]) == self::END_PROCESS_STR) {
359 $result = file($resultFile);
360 break;
361 }
362 }
363 if ($maxtime < time() && !$noTimeout) {
364 break;
365 }
366 }
367
368 $err = file_get_contents($errFile);
369 if (!empty($err)) {
370 Util::logError('VBS error on ' . $basename . ': ' . $err);
371 }
372
373 self::writeLog('Exec ' . $basename . ':');
374 self::writeLog('-> content: ' . str_replace(PHP_EOL, ' \\\\ ', $content));
375 self::writeLog('-> errFile: ' . $errFile);
376 self::writeLog('-> checkFile: ' . $checkFile);
377 self::writeLog('-> resultFile: ' . $resultFile);
378 self::writeLog('-> scriptPath: ' . $scriptPath);
379
380 if ($result !== false && !empty($result)) {
381 $rebuildResult = array();
382 foreach ($result as $row) {
383 $row = trim($row);
384 if (!empty($row)) {
385 $rebuildResult[] = $row;
386 }
387 }
388 $result = $rebuildResult;
389 self::writeLog('-> result: ' . substr(implode(' \\\\ ', $result), 0, 2048));
390 } else {
391 self::writeLog('-> result: N/A');
392 }
393
394 return $result;
395 }
