88 private function createSymlink()
89 {
90 $src = Util::formatWindowsPath($this->currentPath);
91 $dest = Util::formatWindowsPath($this->symlinkPath);
92
93 if(file_exists($dest)) {
94 if (is_link($dest)) {
95 $target = readlink($dest);
96 if ($target == $src) {
97 return;
98 }
99 Batch::removeSymlink($dest);
100 } elseif (is_file($dest)) {
101 Util::logError('Removing . ' . $this->symlinkPath . ' file. It should not be a regular file');
102 unlink($dest);
103 } elseif (is_dir($dest)) {
104 if (!(new \FilesystemIterator($dest))->valid()) {
105 rmdir($dest);
106 } else {
107 Util::logError($this->symlinkPath . ' should be a symlink to ' . $this->currentPath . '. Please remove this dir and restart bearsampp.');
108 return;
109 }
110 }
111 }
112
113 Batch::createSymlink($src, $dest);
