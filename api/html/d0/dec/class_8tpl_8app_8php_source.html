98 public static function getActionRun($action, $args = array(), $item = array(), $waitUntilTerminated = true)
99 {
100 global $bearsamppRoot, $bearsamppCore;
101 $args = $args == null ? array() : $args;
102
103 $argImp = '';
104 foreach ($args as $arg) {
105 $argImp .= ' ' . base64_encode($arg);
106 }
107
108 $result = 'Action: run; ' .
109 'FileName: "' . $bearsamppCore->getPhpExe(true) . '"; ' .
110 'Parameters: "' . Core::isRoot_FILE . ' ' . $action . $argImp . '"; ' .
111 'WorkingDir: "' . $bearsamppRoot->getCorePath(true) . '"';
112
113 if (!empty($item)) {
114 $result = 'Type: item; ' . $result .
115 '; Caption: "' . $item[self::ITEM_CAPTION] . '"' .
116 (!empty($item[self::ITEM_GLYPH]) ? '; Glyph: "' . $item[self::ITEM_GLYPH] . '"' : '');
117 } elseif ($waitUntilTerminated) {
118 $result .= '; Flags: waituntilterminated';
119 }
120
121 return $result;
122 }
