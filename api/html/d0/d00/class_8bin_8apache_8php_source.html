126 protected function replaceAll($params) {
127 $content = file_get_contents($this->bearsamppConf);
128
129 foreach ($params as $key => $value) {
130 $content = preg_replace('|' . $key . ' = .*|', $key . ' = ' . '"' . $value.'"', $content);
131 $this->bearsamppConfRaw[$key] = $value;
132 switch ($key) {
133 case self::LOCAL_CFG_PORT:
134 $this->port = $value;
135 break;
136 case self::LOCAL_CFG_SSL_PORT:
137 $this->sslPort = $value;
138 break;
139 }
140 }
141
142 file_put_contents($this->bearsamppConf, $content);

201 );
