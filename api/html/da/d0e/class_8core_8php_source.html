461 /**
462 * Extracts a 7z file to the specified destination.
463 *
464 * This function uses the `node-7z-archive` library to extract the contents of a 7z file.
465 * It also updates a progress bar to show the extraction progress.
466 *
467 * @param string $filePath The path to the 7z file.
468 * @param string $destination The destination directory.
469 * @return bool True on success, false on failure.
470 * @throws Exception If the extraction fails.
471 */
472 public function unzipFile($filePath, $destination, $progressCallback = null)
473{
474 global $bearsamppRoot;
475
476 $nodePath = $this->getLibsPath() . '/nodejs/node.exe'; // Ensure 'node' is in your PATH
477 $unzipScriptPath = $this->getLibsPath() . '/nodejs/unzip.js';
478
479 if (!file_exists($unzipScriptPath)) {
480 Util::logError('unzip.js not found at: ' . $unzipScriptPath);
481 return false;
482 }
483
484 $command = escapeshellarg($nodePath) . ' ' . escapeshellarg($unzipScriptPath) . ' ' . escapeshellarg($filePath) . ' ' . escapeshellarg($destination);
485 Util::logDebug('Executing command: ' . $command);
486
487 $descriptorspec = [
488 1 => ['pipe', 'w'],
489 2 => ['pipe', 'w']
490 ];
491
492 $process = proc_open($command, $descriptorspec, $pipes);
493
494 if (is_resource($process)) {
495 $lastProgress = 0;
496 while ($line = fgets($pipes[1])) {
497 Util::logDebug('Command output: ' . $line);
498
499 if ($progressCallback) {
500 $data = json_decode($line, true);
501 if (isset($data['progress'])) {
502 $progress = (int) $data['progress'];
503 Util::logDebug('Parsed progress: ' . $progress);
504 if ($progress > $lastProgress) {
505 $lastProgress = $progress;
506 call_user_func($progressCallback, $progress);
507
508 if (ob_get_length()) {
509 ob_flush();
510 }
511 flush();
512 }
513 } elseif (isset($data['success'])) {
514 Util::logDebug('Successfully unzipped file to: ' . $destination);
515 fclose($pipes[1]);
516 fclose($pipes[2]);
517 proc_close($process);
518 return true;
519 } elseif (isset($data['error'])) {
520 Util::logError('Failed to unzip file. Error: ' . $data['error']);
521 fclose($pipes[1]);
522 fclose($pipes[2]);
523 proc_close($process);
524 return false;
525 }
526 }
527 }
528
529 // Check if the process terminated unexpectedly
530 $status = proc_get_status($process);
531 if (!$status['running']) {
532 Util::logError('Process terminated unexpectedly.');
533 fclose($pipes[1]);
534 fclose($pipes[2]);
535 proc_close($process);
536 return false;
537 }
538
539 fclose($pipes[1]);
540 fclose($pipes[2]);
541 proc_close($process);
542 } else {
543 Util::logError('Failed to open process.');
544 return false;
545 }
546}

548 /**
549 * Fetches a file from a given URL and saves it to a specified file path.
550 *
551 * This method attempts to retrieve the content from the provided URL and save it to the specified file path.
552 * If any error occurs during fetching or saving, it logs the error and returns an error message.
553 * If the operation is successful, it returns the file path.
554 * The method also logs the file size if the input stream is a valid resource.
555 *
556 * @param string $moduleUrl The URL from which to fetch the file content.
557 * @param string $filePath The path where the file content should be saved.
558 * @param bool $progressBar Optional. Whether to display a progress bar during the download process. Default is false.
559 *
560 * @return array Returns the file path if successful, or an array with an error message if an error occurs.
561 */
562 public function getFileFromUrl(string $moduleUrl, string $filePath, $progressBar = false)
563 {
564 // Open the URL for reading
565 $inputStream = @fopen( $moduleUrl, 'rb' );
566 if ( $inputStream === false ) {
567 Util::logError( 'Error fetching content from URL: ' . $moduleUrl );
568
569 return ['error' => 'Error fetching module'];
570 }
571
572 // Open the file for writing
573 $outputStream = @fopen( $filePath, 'wb' );
574 if ( $outputStream === false ) {
575 Util::logError( 'Error opening file for writing: ' . $filePath );
576 fclose( $inputStream );
577
578 return ['error' => 'Error saving module'];
579 }
580
581 // Read and write in chunks to avoid memory overload
582 $bufferSize = 8096; // 8KB
583 $chunksRead = 0;
584
585 while ( !feof( $inputStream ) ) {
586 $buffer = fread( $inputStream, $bufferSize );
587 fwrite( $outputStream, $buffer );
588 $chunksRead++;
589
590 // Send progress update
591 if ( $progressBar ) {
592 $progress = $chunksRead;
593 echo json_encode( ['progress' => $progress] );
594
595 // Check if output buffering is active before calling ob_flush()
596 if ( ob_get_length() !== false ) {
597 ob_flush();
598 }
599 flush();
600 }
601 }
602
603 fclose( $inputStream );
604 fclose( $outputStream );
605
606 return ['success' => true];
607 }
