31 public static function getMenuMariadb()
32 {
33 global $bearsamppBins, $bearsamppLang, $bearsamppTools;
34 $resultItems = $resultActions = '';

36 $isEnabled = $bearsamppBins->getMariadb()->isEnable();

45
46 // Enable

65 $resultItems .= $tplService[TplApp::SECTION_CALL] . PHP_EOL;
66 $resultActions .= $tplService[TplApp::SECTION_CONTENT] . PHP_EOL;

70 $resultItems .= $tplDebug[TplApp::SECTION_CALL] . PHP_EOL;
71 $resultActions .= $tplDebug[TplApp::SECTION_CONTENT];
72

75 $bearsamppLang->getValue(Lang::CONSOLE),

79
80 // Conf
81 $resultItems .= TplAestan::getItemNotepad(basename($bearsamppBins->getMariadb()->getConf()), $bearsamppBins->getMariadb()->getConf()) . PHP_EOL;

83 // Errors log

129 public static function getMenuMariadbService()
130 {
131 global $bearsamppLang, $bearsamppBins;
132
133 $tplChangePort = TplApp::getActionMulti(
134 self::ACTION_CHANGE_PORT, null,
135 array($bearsamppLang->getValue(Lang::MENU_CHANGE_PORT), TplAestan::GLYPH_NETWORK),
136 false, get_called_class()
137 );
138
139 $isInstalled = $bearsamppBins->getMariadb()->getService()->isInstalled();
140
141 $result = TplAestan::getItemActionServiceStart($bearsamppBins->getMariadb()->getService()->getName()) . PHP_EOL .
142 TplAestan::getItemActionServiceStop($bearsamppBins->getMariadb()->getService()->getName()) . PHP_EOL .
143 TplAestan::getItemActionServiceRestart($bearsamppBins->getMariadb()->getService()->getName()) . PHP_EOL .
144 TplAestan::getItemSeparator() . PHP_EOL .
145 TplApp::getActionRun(
146 Action::CHECK_PORT, array($bearsamppBins->getMariadb()->getName(), $bearsamppBins->getMariadb()->getPort()),
147 array(sprintf($bearsamppLang->getValue(Lang::MENU_CHECK_PORT), $bearsamppBins->getMariadb()->getPort()), TplAestan::GLYPH_LIGHT)
148 ) . PHP_EOL .
149 $tplChangePort[TplApp::SECTION_CALL] . PHP_EOL;
150
151 $tplChangeRootPwd = null;
152 if ($isInstalled) {
153 $tplChangeRootPwd = TplApp::getActionMulti(
154 self::ACTION_CHANGE_ROOT_PWD, null,
155 array($bearsamppLang->getValue(Lang::MENU_CHANGE_ROOT_PWD), TplAestan::GLYPH_PASSWORD),
156 !$isInstalled, get_called_class()
157 );
158
159 $result .= $tplChangeRootPwd[TplApp::SECTION_CALL] . PHP_EOL;
160 }
161
162 if (!$isInstalled) {
163 $tplInstallService = TplApp::getActionMulti(
164 self::ACTION_INSTALL_SERVICE, null,
165 array($bearsamppLang->getValue(Lang::MENU_INSTALL_SERVICE), TplAestan::GLYPH_SERVICE_INSTALL),
166 $isInstalled, get_called_class()
167 );
168
169 $result .= $tplInstallService[TplApp::SECTION_CALL] . PHP_EOL . PHP_EOL .
170 $tplInstallService[TplApp::SECTION_CONTENT] . PHP_EOL;
171 } else {
172 $tplRemoveService = TplApp::getActionMulti(
173 self::ACTION_REMOVE_SERVICE, null,
174 array($bearsamppLang->getValue(Lang::MENU_REMOVE_SERVICE), TplAestan::GLYPH_SERVICE_REMOVE),
175 !$isInstalled, get_called_class()
176 );
177
178 $result .= $tplRemoveService[TplApp::SECTION_CALL] . PHP_EOL . PHP_EOL .
179 $tplRemoveService[TplApp::SECTION_CONTENT] . PHP_EOL;
180 }
181
182 $result .= $tplChangePort[TplApp::SECTION_CONTENT] . PHP_EOL .
183 ($tplChangeRootPwd != null ? $tplChangeRootPwd[TplApp::SECTION_CONTENT] . PHP_EOL : '');
184
185 return $result;
186 }
