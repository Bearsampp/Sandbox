151 public function processWindow($window, $id, $ctrl, $param1, $param2)
152 {
153 global $bearsamppCore, $bearsamppLang, $bearsamppBins, $bearsamppWinbinder;
154
155 if ($this->version == $this->currentVersion) {
156 $bearsamppWinbinder->messageBoxWarning(sprintf($bearsamppLang->getValue(Lang::SWITCH_VERSION_SAME_ERROR), $this->bin->getName(), $this->version), $this->boxTitle);
157 $bearsamppWinbinder->destroyWindow($window);
158 }
159
160 // scan folder
161 $this->bearsamppSplash->incrProgressBar();
162 if (!empty($this->pathsToScan)) {
163 Util::changePath(Util::getFilesToScan($this->pathsToScan));
164 }
165
166 // switch
167 $this->bearsamppSplash->incrProgressBar();
168 if ($this->bin->switchVersion($this->version, true) === false) {
169 $this->bearsamppSplash->incrProgressBar(self::GAUGE_SERVICES * count($bearsamppBins->getServices()) + self::GAUGE_OTHERS);
170 $bearsamppWinbinder->destroyWindow($window);
171 }
172
173 // stop service
174 if ($this->service != null) {
175 $binName = $this->bin->getName() == $bearsamppLang->getValue(Lang::PHP) ? $bearsamppLang->getValue(Lang::APACHE) : $this->bin->getName();
176 $this->bearsamppSplash->setTextLoading(sprintf($bearsamppLang->getValue(Lang::STOP_SERVICE_TITLE), $binName));
177 $this->bearsamppSplash->incrProgressBar();
178 $this->service->stop();
179 } else {
180 $this->bearsamppSplash->incrProgressBar();
181 }
182
183 // reload config
184 $this->bearsamppSplash->setTextLoading($bearsamppLang->getValue(Lang::SWITCH_VERSION_RELOAD_CONFIG));
185 $this->bearsamppSplash->incrProgressBar();
186 Root::loadConfig();
187
188 // reload bins
189 $this->bearsamppSplash->setTextLoading($bearsamppLang->getValue(Lang::SWITCH_VERSION_RELOAD_BINS));
190 $this->bearsamppSplash->incrProgressBar();
191 $bearsamppBins->reload();
192
193 // change port
194 if ($this->changePort) {
195 $this->bin->reload();
196 $this->bin->changePort($this->bin->getPort());
197 }
198
199 // start service
200 if ($this->service != null) {
201 $binName = $this->bin->getName() == $bearsamppLang->getValue(Lang::PHP) ? $bearsamppLang->getValue(Lang::APACHE) : $this->bin->getName();
202 $this->bearsamppSplash->setTextLoading(sprintf($bearsamppLang->getValue(Lang::START_SERVICE_TITLE), $binName));
203 $this->bearsamppSplash->incrProgressBar();
204 $this->service->start();

206 $this->bearsamppSplash->incrProgressBar();
207 }

209 $this->bearsamppSplash->incrProgressBar(self::GAUGE_SERVICES * count($bearsamppBins->getServices()) + 1);
210 $bearsamppWinbinder->messageBoxInfo(
