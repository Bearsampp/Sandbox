175function raw_db_list_table_fields_def($tablename, $type = false)
176{
177 $result = mysql_query("SHOW COLUMNS FROM $tablename");
178 if ($result === false) return false;
179 $coltype = array();
180 $colnames = array();
181 if (mysql_num_rows($result) > 0) {
182 while ($row = mysql_fetch_assoc($result)) {
183 $colnames[] = $row['Field'];
184 $coltype[] = $row['Type'];
185 } // while
186 }
187 if (mysql_free_result($result) == false) return false;
188 return ($type ? $coltype : $colnames);
189}
