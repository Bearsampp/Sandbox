62 if (!$this->enable) {
63 Util::logInfo($this->name . ' is not enabled!');

107 /**
108 * Updates the module configuration with a specific version.
109 *
110 * @param string|null $version The version to update to. If null, the current version is used.
111 * @param int $sub The sub-level for logging indentation.
112 * @param bool $showWindow Whether to show a window during the update process.

115 protected function updateConfig($version = null, $sub = 0, $showWindow = false) {
116 global $bearsamppLang, $bearsamppWinbinder;
117
118 if (!$this->enable) {
119 return true;
120 }
121
122 $version = $version == null ? $this->version : $version;
123 Util::logDebug(($sub > 0 ? str_repeat(' ', 2 * $sub) : '') . 'Update ' . $this->name . ' ' . $version . ' config');
124
125 $boxTitle = sprintf($bearsamppLang->getValue(Lang::SWITCH_VERSION_TITLE), $this->getName(), $version);
126
127 $conf = str_replace('nodejs' . $this->getVersion(), 'nodejs' . $version, $this->getConf());
128 $bearsamppConf = str_replace('nodejs' . $this->getVersion(), 'nodejs' . $version, $this->bearsamppConf);
129
130 if (!file_exists($conf) || !file_exists($bearsamppConf)) {
131 Util::logError('bearsampp config files not found for ' . $this->getName() . ' ' . $version);
132 if ($showWindow) {
133 $bearsamppWinbinder->messageBoxError(
134 sprintf($bearsamppLang->getValue(Lang::BEARSAMPP_CONF_NOT_FOUND_ERROR), $this->getName() . ' ' . $version),
135 $boxTitle
136 );
137 }
138 return false;
139 }
140
141 $bearsamppConfRaw = parse_ini_file($bearsamppConf);
142 if ($bearsamppConfRaw === false || !isset($bearsamppConfRaw[self::ROOT_CFG_VERSION]) || $bearsamppConfRaw[self::ROOT_CFG_VERSION] != $version) {
143 Util::logError('bearsampp config file malformed for ' . $this->getName() . ' ' . $version);
144 if ($showWindow) {
145 $bearsamppWinbinder->messageBoxError(
146 sprintf($bearsamppLang->getValue(Lang::BEARSAMPP_CONF_MALFORMED_ERROR), $this->getName() . ' ' . $version),
