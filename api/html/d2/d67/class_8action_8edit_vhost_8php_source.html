68 public function processWindow($window, $id, $ctrl, $param1, $param2)

111 $bearsamppWinbinder->resetProgressBar($this->wbProgressBar);

120 $bearsamppWinbinder->incrProgressBar($this->wbProgressBar);

128 $bearsamppWinbinder->destroyWindow($window);
129 } else {
